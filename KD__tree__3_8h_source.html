28  /* Public interface for building kdtree.
29  * Give points should have format equivalent to double[npnts][3].
30  * It also takes an optional argument maxn to specify the maximum
31  * number of points that can be returned by search.
32  */
33  void build( const double *pnts, int np, int maxn=0) {
34  assert(np>=0);
35  _npnts = np;
36  if (maxn) _maxout = maxn; else _maxout = np;
37 
38  if (_npnts) {
39  _tree = new int[2*_npnts-1];
40  _bboxes = new double[12*_npnts-6];
41 
42  kdtree_build_3( pnts, _npnts);
43  /* Compute bboxes in linear time */
44  kdtree_bboxes_3( pnts);
45  }
46  }

48  /* Public interface for range search.
49  * range is specified as {minx, miny, minz, maxx, maxy, maxz}.
50  * If indices is given as second argument, then point to buffer space
51  * for indices to the points is returned.

54  int search( const double range[6], int **indices=0, int start=0) {

64  /* Public interface for range search.
65  * range is specified as {minx, miny, minz, maxx, maxy, maxz}.
66  * If indices is given as second argument, then point to buffer space
67  * for indices to the points is returned.
68  * This function must be called after build has been called.
69  */
70  int search( const double pnt[3], const double tol, int **indices=0, int start=0) {
71  double range[6];
72 
73  for ( int k=0; k<3; ++k) {
74  range[k] = pnt[k] - tol;
75  range[3+k] = pnt[k] + tol;
76  }
77 
78  if (!_indices) _indices = new int[_maxout];
79  if (indices) *indices = _indices;
80 
81  return kdtree_search_3( range, start);
82  }
