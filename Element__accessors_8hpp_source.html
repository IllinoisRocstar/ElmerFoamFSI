209 template <class Value>
210 class Element_node_vectors_k_const {
211  public:
212  enum { MAX_NODES=9};
213  typedef unsigned int Size;
214 
216  Element_node_vectors_k_const() : _attr(NULL)
217  { std::fill_n( _vs, 9, reinterpret_cast<Value*>(NULL)); }
218 
220  void set( const Value *p, Element_node_enumerator &ene, int strd) {
221  COM_assertion( ene.size_of_nodes() <= MAX_NODES);
222  _attr = NULL;
223  for ( int i=ene.size_of_nodes()-1; i>=0; --i)
224  _vs[i] = &p[(ene[i]-1)*strd];
225  }
226 
228  void set( const DataItem *a, Element_node_enumerator &ene) {
229  COM_assertion( a->is_nodal() && a->pane()->id());
230  COM_assertion( sizeof(Value)==COM_get_sizeof( a->data_type(),1));
231 
232  int ncomp = a->size_of_components();
233  int strd = a->stride();
234  int nn = ene.size_of_nodes();
235  if ( ncomp<=strd) {
236  const Value *p = reinterpret_cast<const Value*>(a->pointer());
237  _attr = NULL;
238  for ( int i=0; i<nn; ++i) _vs[i] = &p[(ene[i]-1)*strd];
239  }
240  else {
241  _attr = a;
242  for ( int i=0; i<nn; ++i) _offsets[i] = ene[i]-1;
243  }
244  }
245 
246  // Access a vector.
247  const Value &operator()( int i, int j) const {
248  if ( _attr) {
249  COM_assertion( _attr->size_of_components()>1);
250  const DataItem *aj = _attr+j+1;
251  return reinterpret_cast<const Value*>
252  (aj->pointer())[_offsets[i]*aj->stride()];
253  }
254  else
255  return _vs[i][j];
256  }
257 
258  const Value &operator()(int i) const { return operator()(i,0); }
259 
260  const Value &operator[](int i) const { return operator()(i,0); }
261 
262 protected:
263  const DataItem *_attr;
264  union {
265  int _offsets[MAX_NODES]; // Used if _attr is not NULL
266  const Value *_vs[MAX_NODES]; // Used if _attr is NULL
267  };
268 };
